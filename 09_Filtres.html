  this.array = [
    {name: 'Tobias'},

    {name: 'Brian'}
  ];
  this.filteredArray = filterFilter(this.array, 'a');
}]);

Créer un filtre


app.filter('monfiltre', function() {
  return function(input, uppercase) {
    input = input || '';
    var out = "";
    for (var i = 0; i < input.length; i++) {
      out = input.charAt(i) + out;
    }
    if (uppercase) {
      out = out.toUpperCase();
    }
    return out;
  };
})
